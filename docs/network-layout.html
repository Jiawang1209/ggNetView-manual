Build graph object

graph_obj <- build_graph_from_mat(
  mat = otu_rare_relative,
  transfrom.method = "none", #  based your input data
  r.threshold = 0.7,
  p.threshold = 0.05,
  method = "WGCNA",
  cor.method = "pearson",
  proc = "BH",
  module.method = "Fast_greedy",
  node_annotation = tax_tab,
  top_modules = 15,
  seed = 1115
)

p <- ggNetView(
  graph_obj = graph_obj,
  layout = "gephi",
  center = F,
  shrink = 0.8,
  layout.module = "random",
  group.by = "Modularity",
  fill.by = "Modularity"
